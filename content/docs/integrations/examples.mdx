## Prerequisites

All examples require:
- **API Key** — Generate from dashboard `/settings/api-keys`
- **Base URL** — `https://api.emberly.ca`
- **Headers** — Include `Authorization: Bearer YOUR_API_KEY` and `Content-Type: application/json`

### Basic Setup

```javascript
const axios = require('axios');

const api = axios.create({
  baseURL: 'https://api.emberly.ca',
  headers: {
    'Authorization': `Bearer ${process.env.EMBERLY_API_KEY}`,
    'Content-Type': 'application/json'
  }
});

// Handle errors consistently
api.interceptors.response.use(
  response => response.data,
  error => {
    const message = error.response?.data?.error || error.message;
    throw new Error(`API Error: ${message}`);
  }
);

module.exports = api;
```

### Example 1: Upload a File

```javascript
const FormData = require('form-data');
const fs = require('fs');
const axios = require('axios');

async function uploadFile() {
  const formData = new FormData();
  formData.append('file', fs.createReadStream('document.pdf'));
  formData.append('name', 'My Document');
  formData.append('expiresIn', '30d'); // expires in 30 days

  try {
    const response = await axios.post(
      'https://api.emberly.ca/files/upload',
      formData,
      {
        headers: {
          ...formData.getHeaders(),
          'Authorization': `Bearer ${process.env.EMBERLY_API_KEY}`
        },
        onUploadProgress: (progressEvent) => {
          const percentComplete = Math.round(
            (progressEvent.loaded * 100) / progressEvent.total
          );
          console.log(`Upload progress: ${percentComplete}%`);
        }
      }
    );

    console.log('File uploaded:', response.data);
    return response.data;
  } catch (error) {
    console.error('Upload failed:', error.response?.data || error.message);
    throw error;
  }
}

uploadFile();
```

### Example 3: Download a File from URL

```javascript
const axios = require('axios');
const fs = require('fs');

async function downloadFile(fileUrl, outputPath) {
  try {
    const response = await axios.get(fileUrl, {
      responseType: 'stream',
      headers: {
        'Authorization': `Bearer ${process.env.EMBERLY_API_KEY}`
      }
    });

    const writer = fs.createWriteStream(outputPath);
    response.data.pipe(writer);

    return new Promise((resolve, reject) => {
      writer.on('finish', resolve);
      writer.on('error', reject);
    });
  } catch (error) {
    console.error('Download failed:', error.message);
    throw error;
  }
}

downloadFile('https://file.emberly.ca/files/abc123', './downloaded.pdf')
  .then(() => console.log('Download complete'))
  .catch(error => console.error(error));
```

### Basic Setup

```python
import requests
import os
from dotenv import load_dotenv

load_dotenv()

class EmberyAPI:
    def __init__(self):
        self.base_url = 'https://api.emberly.ca'
        self.headers = {
            'Authorization': f"Bearer {os.getenv('EMBERLY_API_KEY')}",
            'Content-Type': 'application/json'
        }

    def request(self, method, endpoint, **kwargs):
        url = f"{self.base_url}{endpoint}"
        response = requests.request(method, url, headers=self.headers, **kwargs)
        response.raise_for_status()
        return response.json()

    def get(self, endpoint, **kwargs):
        return self.request('GET', endpoint, **kwargs)

    def post(self, endpoint, **kwargs):
        return self.request('POST', endpoint, **kwargs)

    def delete(self, endpoint, **kwargs):
        return self.request('DELETE', endpoint, **kwargs)

api = EmberyAPI()
```

### Example 1: Upload a File

```python
import requests
from pathlib import Path

def upload_file(file_path, name=None, expires_in='30d'):
    url = 'https://api.emberly.ca/files/upload'
    headers = {
        'Authorization': f"Bearer {os.getenv('EMBERLY_API_KEY')}"
    }
    
    with open(file_path, 'rb') as f:
        files = {'file': (name or Path(file_path).name, f)}
        data = {
            'name': name or Path(file_path).name,
            'expiresIn': expires_in
        }
        
        response = requests.post(url, files=files, data=data, headers=headers)
        response.raise_for_status()
        
        file_data = response.json()
        print(f"Uploaded: {file_data['name']}")
        print(f"URL: {file_data['url']}")
        return file_data

# Usage
upload_file('document.pdf', name='My Document')
```

### Example 3: List and Filter Files

```python
from api import api
from datetime import datetime, timedelta

def list_files_with_filters():
    # Get all files
    response = api.get('/files')
    files = response.get('data', [])
    
    # Filter by date (modified in last 7 days)
    week_ago = datetime.now() - timedelta(days=7)
    recent_files = [
        f for f in files 
        if datetime.fromisoformat(f['updatedAt']) > week_ago
    ]
    
    print(f"Recent files ({len(recent_files)}):")
    for file in recent_files:
        size_mb = file['size'] / (1024 * 1024)
        print(f"  - {file['name']} ({size_mb:.2f} MB)")
    
    # Filter by type
    images = [f for f in files if f['type'].startswith('image/')]
    documents = [f for f in files if f['type'] in ['application/pdf', 'application/msword']]
    
    print(f"\nImages: {len(images)}")
    print(f"Documents: {len(documents)}")
    
    return recent_files

list_files_with_filters()
```

### Basic Setup

```go
package main

import (
  "github.com/go-resty/resty/v2"
  "os"
)

type EmberyClient struct {
  client *resty.Client
}

func NewEmberyClient() *EmberyClient {
  client := resty.New().
    SetBaseURL("https://api.emberly.ca").
    SetHeader("Authorization", "Bearer " + os.Getenv("EMBERLY_API_KEY")).
    SetHeader("Content-Type", "application/json")
  
  return &EmberyClient{client: client}
}

func (e *EmberyClient) ListFiles() ([]map[string]interface{}, error) {
  var files []map[string]interface{}
  _, err := e.client.R().
    SetResult(&files).
    Get("/files")
  
  return files, err
}
```

### Example 1: Upload File

```go
package main

import (
  "fmt"
  "os"
)

func (e *EmberyClient) UploadFile(filePath string, name string) error {
  file, err := os.Open(filePath)
  if err != nil {
    return err
  }
  defer file.Close()

  resp, err := e.client.R().
    SetFileReader("file", name, file).
    SetFormData(map[string]string{
      "name": name,
      "expiresIn": "30d",
    }).
    Post("/files/upload")

  if err != nil {
    return err
  }

  fmt.Printf("Upload response: %v\n", resp)
  return nil
}

// Usage
func main() {
  client := NewEmberyClient()
  err := client.UploadFile("document.pdf", "My Document")
  if err != nil {
    panic(err)
  }
}
```

## Bash / cURL

### Basic Pattern

```bash
#!/bin/bash

API_KEY="${EMBERLY_API_KEY}"
BASE_URL="https://api.emberly.ca"

# Helper function
emberly_api() {
  local method=$1
  local endpoint=$2
  local data=$3

  local args=(
    -X "$method"
    -H "Authorization: Bearer $API_KEY"
    -H "Content-Type: application/json"
  )

  if [ -n "$data" ]; then
    args+=(-d "$data")
  fi

  curl "${args[@]}" "$BASE_URL$endpoint"
}
```

## Error Handling Examples

### JavaScript

```javascript
async function handleErrors() {
  try {
    const response = await api.post('/files/upload', data);
  } catch (error) {
    if (error.response?.status === 401) {
      console.error('Authentication failed. Check your API key.');
    } else if (error.response?.status === 403) {
      console.error('Permission denied.');
    } else if (error.response?.status === 429) {
      console.error('Rate limited. Wait before retrying.');
      // Implement exponential backoff
    } else if (error.response?.status === 500) {
      console.error('Server error. Try again later.');
    } else {
      console.error(error.response?.data?.error || error.message);
    }
  }
}
```

---

### Python

```python
def handle_errors(func):
  def wrapper(*args, **kwargs):
    try:
      return func(*args, **kwargs)
    except requests.exceptions.HTTPError as e:
      status = e.response.status_code
      if status == 401:
        print("Authentication failed")
      elif status == 403:
        print("Permission denied")
      elif status == 429:
        print("Rate limited")
      else:
        print(f"Error: {e.response.json()['error']}")
  return wrapper

@handle_errors
def api_call():
  response = requests.get(...)
  response.raise_for_status()
```

## Rate Limiting

All API calls are rate-limited to prevent abuse.

**Limits:**
- **Authenticated requests:** 1,000 requests per hour
- **Unauthenticated:** 100 requests per hour
- **Upload endpoint:** 100 uploads per hour

**Rate limit headers:**
```
X-RateLimit-Limit: 1000
X-RateLimit-Remaining: 987
X-RateLimit-Reset: 1234567890
```

**Handling rate limits:**
```javascript
// Extract retry-after
const resetTime = parseInt(response.headers['X-RateLimit-Reset']) * 1000;
const waitMs = resetTime - Date.now();

// Wait and retry
setTimeout(() => {
  retryRequest();
}, waitMs);
```
